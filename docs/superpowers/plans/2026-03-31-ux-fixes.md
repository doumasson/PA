async def _filter_already_notified(store, emails: list[dict]) -> list[dict]:
    """Remove emails that were already notified about."""
    rows = await store.fetchall("SELECT message_id FROM google_notified_emails")
    seen = {r['message_id'] for r in rows}
    return [e for e in emails if e['id'] not in seen]


async def _record_notified_emails(store, message_ids: list[str]) -> None:
    """Record email IDs that we just notified about."""
    for mid in message_ids:
        await store.execute(
            "INSERT OR IGNORE INTO google_notified_emails (message_id) VALUES (?)",
            (mid,),
        )


async def _cleanup_old_notifications(store) -> None:
    """Delete notification records older than 14 days."""
    await store.execute(
        "DELETE FROM google_notified_emails WHERE notified_at < datetime('now', '-14 days')"
    )


async def _mark_as_read(gmail_service, message_ids: list[str]) -> None:
    """Mark emails as read in Gmail."""
    for mid in message_ids:
        try:
            gmail_service.users().messages().modify(
                userId='me', id=mid, body={'removeLabelIds': ['UNREAD']}
            ).execute()
        except Exception:
            pass

```python
async def handle_kid_sport(ctx: AppContext, text: str, update: Update) -> str:
    """Update what sport a kid plays. Uses Claude to parse natural corrections."""
    from pa.core.tier import Tier
    import json, re

    PARSE = """Extract kid-sport assignments from this message. Steven has two sons:
- Maddox (12)
- Asher (10)

Return ONLY a JSON array of objects: [{"kid": "maddox"|"asher", "sport": "sport_name"}]
Examples:
- "Asher is the soccer player, not Maddox" → [{"kid": "asher", "sport": "soccer"}]
- "Asher is soccer, Maddox is basketball" → [{"kid": "asher", "sport": "soccer"}, {"kid": "maddox", "sport": "basketball"}]
- "Maddox plays basketball" → [{"kid": "maddox", "sport": "basketball"}]
Raw JSON array only."""

    result = await ctx.brain.query(text, system_prompt=PARSE, tier=Tier.FAST, use_conversation=False)
    try:
        result = re.sub(r',\s*([}\]])', r'\1', result.strip())
        start = result.find('[')
        end = result.rfind(']')
        if start == -1:
            raise ValueError("No JSON array found")
        assignments = json.loads(result[start:end + 1])
    except Exception:
        # Fallback to simple keyword parsing
        tl = text.lower()
        kid = None
        sport = None
        if "maddox" in tl:
            kid = "maddox"
        elif "asher" in tl:
            kid = "asher"
        sports = ["basketball", "football", "soccer", "baseball", "hockey",
                  "lacrosse", "swimming", "tennis", "golf", "volleyball",
                  "wrestling", "track", "cross country", "gymnastics"]
        for s in sports:
            if s in tl:
                sport = s
                break
        if kid and sport:
            assignments = [{"kid": kid, "sport": sport}]
        else:
            return "I couldn't figure out which kid plays what. Try: 'Asher plays soccer' or 'Maddox does basketball'."

    if not assignments:
        return "I couldn't figure out which kid plays what. Try: 'Asher plays soccer'."

    confirmations = []
    for a in assignments:
        kid = a.get("kid", "").lower()
        sport = a.get("sport", "").lower()
        if kid not in ("maddox", "asher") or not sport:
            continue
        await ctx.store.execute(
            "INSERT INTO google_state (key, value) VALUES (?, ?) "
            "ON CONFLICT(key) DO UPDATE SET value=excluded.value",
            (f"kid_{kid}_sport", sport),
        )
        await ctx.brain.learn_preference(
            f"{kid.capitalize()} plays {sport.upper()} (not other sports)",
            learned_from="kid_sport_update"
        )
        confirmations.append(f"{kid.capitalize()} plays {sport}")

    if not confirmations:
        return "I couldn't figure out which kid plays what. Try: 'Asher plays soccer'."

    return "Got it — " + ", ".join(confirmations) + ". I'll update the email triage."
```

def _format_job_error(job_name: str, error: Exception) -> str:
    """Convert job errors into human-readable Telegram messages."""
    msg = str(error)
    msg_lower = msg.lower()

    # OAuth / token expiry (Gmail)
    if "invalid_grant" in msg_lower or ("token" in msg_lower and ("expired" in msg_lower or "revoked" in msg_lower)):
        if "gmail" in job_name:
            return "Gmail connection expired. Need to reauthorize — run /gmail_auth."
        return "Authentication expired. Need to reconnect — check /status."

    # Teller / bank API errors
    if "401" in msg and "authentication_error" in msg_lower:
        return "Bank connection expired. Use /sync to reconnect."

    # Generic API / network errors
    if any(w in msg_lower for w in ["connection refused", "timeout", "unreachable", "503", "502"]):
        return f"Couldn't reach the service for {job_name}. Will retry next scheduled check."

    # Fallback: generic but clean (no JSON, no tracebacks)
    clean = msg.split("{")[0].split("Traceback")[0].strip()[:150]
    return f"Job {job_name} had an issue: {clean or 'unknown error'}"

[{"description": "original description", "category": "category"}]

Valid categories: Groceries, Gas/Convenience, Food/Dining, Food/Delivery, Food/Coffee,
Liquor, Shopping, Software, Subscription, Bills, Transfer, Recreation, Medical,
Kids, Income, Cash Advance, Unknown

Use "Unknown" ONLY if truly unidentifiable. Never use "general" or "Other".
Raw JSON array only."""

        try:
            result = await brain.query(
                f"Categorize:\n{descs}",
                system_prompt=PROMPT, tier=Tier.FAST, use_conversation=False,
            )
            result = re.sub(r',\s*([}\]])', r'\1', result.strip())
            start = result.find('[')
            end = result.rfind(']')
            if start != -1:
                parsed = json.loads(result[start:end + 1])
                for item in parsed:
                    desc = item.get("description", "")
                    cat = item.get("category", "Unknown")
                    if cat.lower() != "unknown":
                        await learn_category(store, desc, cat, source="ai")
                    results.append(item)
        except Exception:
            pass

    return results

```python
async def save_bills_to_db(ctx, bills: list[dict]) -> int:
    """Save extracted bill data to finance_debts and finance_accounts.
    Returns count of successfully saved bills."""
    import logging
    log = logging.getLogger(__name__)
    saved = 0

    for bill in bills:
        try:
            institution = bill.get('institution', '').strip()
            account_name = bill.get('account_name', '').strip()
            if not institution or not account_name:
                continue

            # Safely convert balance
            try:
                balance = float(bill.get('balance', 0) or 0)
            except (ValueError, TypeError):
                continue
            if balance <= 0:
                continue

            status = bill.get('status', 'current')
            account_type = bill.get('account_type', 'credit_card')
            minimum_payment = bill.get('minimum_payment')
            due_date = bill.get('due_date')
            apr = bill.get('apr')

            if minimum_payment:
                try:
                    minimum_payment = float(minimum_payment)
                except (ValueError, TypeError):
                    minimum_payment = None
            if apr:
                try:
                    apr = float(apr)
                except (ValueError, TypeError):
                    apr = None

            # Check for existing debt to log changes
            existing = await ctx.store.fetchone(
                "SELECT balance FROM finance_debts WHERE institution = ? AND account_name = ?",
                (institution, account_name),
            )
            if existing and abs(float(existing['balance']) - balance) > 0.01:
                log.info("Debt updated: %s %s: $%.2f → $%.2f",
                         institution, account_name, float(existing['balance']), balance)

            # Upsert into finance_debts
            await ctx.store.execute(
                """INSERT INTO finance_debts (institution, account_name, account_type, balance,
                   minimum_payment, apr, due_date, status, updated_at)
                   VALUES (?, ?, ?, ?, ?, ?, ?, ?, CURRENT_TIMESTAMP)
                   ON CONFLICT(institution, account_name) DO UPDATE SET
                   balance=excluded.balance, minimum_payment=excluded.minimum_payment,
                   apr=excluded.apr, due_date=excluded.due_date, status=excluded.status,
                   updated_at=CURRENT_TIMESTAMP""",
                (institution, account_name, account_type, balance,
                 minimum_payment, apr, due_date, status),
            )
            saved += 1
        except Exception as e:
            log.error("save_bills_to_db error for %s: %s", bill.get('institution', '?'), e)

    return saved
```
